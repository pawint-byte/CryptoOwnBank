1. New Functions to Add to shared/financial-math.ts
TypeScriptexport interface HarvestOpportunity {
  assetSymbol: string;
  currentPrice: number;
  quantity: number;
  totalCostBasis: number;
  currentValue: number;
  unrealizedLoss: number;
  estTaxSavings: number; // at user's chosen bracket
  suggestedSwap?: { symbol: string; reason: string };
}

export function scanForHarvestOpportunities(
  positions: PortfolioPosition[],           // from your storage/portfolio query
  priceLookup: Record<string, number>,
  taxBracket: number = 0.32                  // default 32% — let user select
): HarvestOpportunity[] {
  return positions
    .filter(pos => {
      const qty = parseFloat(pos.quantity);
      const value = qty * (priceLookup[pos.assetSymbol] || 0);
      return value < parseFloat(pos.totalCostBasis); // only losses
    })
    .map(pos => {
      const qty = parseFloat(pos.quantity);
      const cost = parseFloat(pos.totalCostBasis);
      const value = qty * (priceLookup[pos.assetSymbol] || 0);
      const loss = cost - value;

      return {
        assetSymbol: pos.assetSymbol,
        currentPrice: priceLookup[pos.assetSymbol] || 0,
        quantity: qty,
        totalCostBasis: cost,
        currentValue: value,
        unrealizedLoss: loss,
        estTaxSavings: loss * taxBracket,
        suggestedSwap: getSwapSuggestion(pos.assetSymbol) // simple map you maintain
      };
    })
    .sort((a, b) => b.estTaxSavings - a.estTaxSavings);
}

// Tiny helper (expand with more pairs later)
function getSwapSuggestion(symbol: string) {
  const swaps: Record<string, { symbol: string; reason: string }> = {
    XRP: { symbol: "USDC", reason: "Stable alternative on XRPL/Stellar" },
    RLUSD: { symbol: "USDC", reason: "Same yield profile" },
    // add more as you grow
  };
  return swaps[symbol];
}
